delete amp_pages from amp_pages p
left outer join amp_reports r on (p.page_name like r.name)
where page_code not in ('DTP','FP') and amp_report_id is null;

delete amp_page_filters from amp_page_filters pf 
left outer join amp_pages p on (pf.amp_page_id=p.amp_page_id)
where p.amp_page_id is null;

delete amp_team_reports
from amp_team_reports tr left outer join amp_reports r 
on (r.amp_report_id=tr.report) where r.amp_report_id is null;

delete from amp_team_page_filters;

insert into amp_team_page_filters 
select ' 'a.amp_team_id,b.amp_page_id,b.amp_filter_id from amp_team a, amp_page_filters b
order by a.amp_team_id,b.amp_page_id,b.amp_filter_id;
